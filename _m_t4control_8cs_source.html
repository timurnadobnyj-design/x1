00227         private List<string> getSymbolList(string filePath)
00228         {
00229             int symbolOffset = 116;
00230             int symbolRecordLength = 128;
00231             int symbolLength = 12;
00232             List<string> symbols = new List<string>();
00233             FileStream fStream = new FileStream(filePath, FileMode.Open, FileAccess.Read);
00234             BinaryReader fReader = new BinaryReader(fStream);
00235             int symbolsNumber = (((int)fStream.Length) - 4) / symbolRecordLength;
00236             fReader.ReadBytes(4);
00237             char[] test;
00238             string symbol;
00239             for (int i = 0; i < symbolsNumber; i++)
00240             {
00241                 test = fReader.ReadChars(symbolLength);
00242                 fReader.ReadBytes(symbolOffset);
00243                 symbol = "";
00244                 foreach (char c in test)
00245                 {
00246                     if (c == '\x00') break;
00247                     symbol = symbol + c.ToString();
00248                 }
00249                 symbols.Add(symbol);
00250             }
00251 
00252             if (fReader != null) fReader.Close();
00253             if (fStream != null) fStream.Close();
00254 
00255             return symbols;
00256         }
