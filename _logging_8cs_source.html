00022 using System;
00023 using System.IO;
00024 
00025 
00026 namespace Skilful
00027 {
00028     public class Logging
00029     {
00030         static StreamWriter sw = null;
00031         
00032         //true to append to existing or create new, false to overwrite existing or create new
00033         public static void Start(string fileName, bool append)
00034         {
00035             sw = new StreamWriter(fileName, append);
00036             sw.AutoFlush = true; //будет сразу в файл сбрасывать, не буферизуя
00037         }
00038 
00039         public static void Log(string description)
00040         {
00041             DateTime dtNow = DateTime.Now;
00042             sw.WriteLine(string.Format("{0:yyyy-MM-dd HH:mm:ss.fff}: {1}", dtNow, description));
00043         }
00044 
00045         public static void Stop()
00046         {
00047             sw.Close();
00048         }
00049     }
00050 }
