00097                 s.IndexOf("OPEN") >= 0    // check for existings of all needed header values
00098             && s.IndexOf("HIGH") >= 0
00099             && s.IndexOf("LOW") >= 0
00100             && s.IndexOf("CLOSE") >= 0
00101             && s.IndexOf("\t") >= 0) // check for column delimiter "\t"
00102             {
00103                 //date parse
00104                 int time = 0, open = 0, high = 0, low = 0, close = 0;
00105                 //get hash of column indexes by headers
00106                 string[] cols = s.Split('\t');
00107                 for (int i = 0; i < cols.Length; i++)
00108                 {
00109                     switch (cols[i])
00110                     {
00111                         case "TIME": time = i; break;
00112                         case "OPEN": open = i; break;
00113                         case "HIGH": high = i; break;

00115                         case "CLOSE": close = i; break;
00116                     }
00117                 }

00119                 string line;
00120                 while ((line = sr.ReadLine()) != null)
00121                 {
00122                     //проверка переполнения
00123                     if ((qi + 5) > quotes.Length) break;
00124                     //проверка диапазона
00125                     if (shift > 0 && cnt >= shift) continue;
00126                     if (len > 0 && len + shift > cnt) break;
00127                     cnt++;
00128                     //сборка бара
00129                     cols = line.Split('\t');
00130                     if (cols.Length < 6) continue;
00131 
00132                     //date
00133                     int year = int.Parse(cols[time].Substring(6, 2));
00134                     if (year < 100) year += year >= 70 ? 1900 : 2000;
00135                     int month = int.Parse(cols[time].Substring(3, 2));
00136                     int day = int.Parse(cols[time].Substring(0, 2));
00137                     //time
00138                     int hour = int.Parse(cols[time].Substring(9, 2)) * 60 * 60;
00139                     int minute = int.Parse(cols[time].Substring(12, 2)) * 60;
00140                     int second = int.Parse(cols[time].Substring(15, 2));
00141                     DateTime cdate = new DateTime(year, month, day);
00142                     DateTime nuldate = new DateTime(1970, 01, 01);
00143                     TimeSpan ts = cdate - nuldate;
00144                     int ttime = ts.Days * 24 * 60 * 60 + hour + minute + second;
00145 
00146                     quotes[qi++] = ttime;

00148                     quotes[qi++] = a2f(cols[high]);
