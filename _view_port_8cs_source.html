00183         public void AutoMarginLooking()
00184         {
00185 
00186             finishTime = (int)linkedSerie.viewPort.VPMaxTime+1;
00187             stratTime =(int) linkedSerie.viewPort.VPMinTime;
00188 
00189             if (finishTime > linkedSerie.data.Count)    finishTime = linkedSerie.data.Count;
00190             if (finishTime < 0) finishTime = 0;
00191 
00192             if (linkedSerie.viewPort.VPMinTime < 0)
00193               stratTime = 0;
00194             if (linkedSerie.viewPort.VPMinTime >= linkedSerie.data.Count)
00195                 stratTime = linkedSerie.data.Count-1;
00196             
00197             
00198             linkedSerie.viewPort.visibleMaxPrice = float.NegativeInfinity;
00199             linkedSerie.viewPort.visibleMinPrice = absRealPriceMax;
00200 
00201             for (int i = stratTime; i < finishTime; i++)
00202             {
00203                 if (i > linkedSerie.viewPort.VPMinTime)
00204                 {
00205                     if (linkedSerie.viewPort.visibleMaxPrice < (float)linkedSerie.data[i].d[3])
00206                     {
00207                         linkedSerie.viewPort.visibleMaxPrice = (float)linkedSerie.data[i].d[3];
00208                        // linkedSerie.viewPort.VisibleMaxPrice.X = i;
00209 
00210                     }
00211                     if (linkedSerie.viewPort.visibleMinPrice > (float)linkedSerie.data[i].d[4])
00212                     {
00213                         linkedSerie.viewPort.visibleMinPrice = (float)linkedSerie.data[i].d[4];
00214                        // linkedSerie.viewPort.VisibleMinPrice.X = i;
00215                     }
00216                 }
00217 
00218             }
00219             if (!float.IsInfinity(margin) && margin != 0 &&  !float.IsNaN(margin) && visibleMaxPrice != 0 && visibleMinPrice != 0)
00220             {
00221                 linkedSerie.viewPort.VPMaxPrice = visibleMaxPrice + margin * visibleMaxPrice;
00222                 linkedSerie.viewPort.VPMinPrice = visibleMinPrice - margin * visibleMinPrice;
00223             }
00224             else
00225             {
00226                 linkedSerie.viewPort.VPMaxPrice = visibleMaxPrice;
00227                 linkedSerie.viewPort.VPMinPrice = visibleMinPrice;
00228             }
00229             linkedSerie.viewPort.varPrice = linkedSerie.viewPort.VPMaxPrice - linkedSerie.viewPort.VPMinPrice;

00233        public void LookForMinMax(Data.Series ds)
00234         {
00235             absRealPriceMax = float.NegativeInfinity;
00236             absRealPriceMin = float.PositiveInfinity;
00237             visibleMaxPrice = float.NegativeInfinity;
00238             visibleMinPrice = float.PositiveInfinity;
00239             //зададим отступ в 5 пунктов
00240             //double dig = (ds.data[0].d[3]-Math.Truncate(ds.data[0].d[3]))*100;
00241             double padding = 5 * (ds.info != null ? ds.info.Symbol.pip : 0.0005);
00242             //внутренний цикл для расчета масштаба(макс-мин) на участве графика, 
00243             //предназначенного для первоначальной отрисовки.
00244             for (int i = 1; i < ds.data.Count; i++)
00245             {
00246                 double lastBarHigh = (double)ds.data[i].d[3],
00247                        lastBarLow = (double)ds.data[i].d[4];
00248                 if ((ds.info!=null? ds.info.log: ds.legend.log))
00249                 {
00250                     lastBarHigh = Math.Log10(Math.Pow(10, lastBarHigh) + padding);
00251                     lastBarLow = Math.Log10(Math.Pow(10, lastBarLow) - padding);
00252                 }
00253                 else{
00254                     lastBarHigh = (double)ds.data[i].d[3] + padding;
00255                     lastBarLow = (double)ds.data[i].d[4] - padding;
00256                 }
00257                 if (i > VPMinTime)
00258                 {
00259                     if (visibleMaxPrice < lastBarHigh)
00260                     {
00261                         visibleMaxPrice = (float)lastBarHigh;
00262                        // VisibleMaxPrice.X = i;
00263                        
00264                     }
00265                     if (visibleMinPrice > lastBarLow)
00266                     {
00267                         visibleMinPrice = (float)lastBarLow;
00268                        // VisibleMinPrice.X = i;
00269                         
00270                     }
00271                 }
00272 
00273 
00274                 if (absRealPriceMax < (float)ds.data[i].d[3])
00275                     absRealPriceMax = (float)ds.data[i].d[3];
00276 
00277                 if (absRealPriceMin > (float)ds.data[i].d[4])
00278                     absRealPriceMin = (float)ds.data[i].d[4];
00279             }
