00380         public bool IsModelClicked(int X, int Y, Data.ViewPort viewPort)
00381         {
00382             X1 = (int) viewPort.BarNumberToPixels(PointsForDrawing_1_3[0].X);
00383             X2 = (int) viewPort.BarNumberToPixels(PointsForDrawing_1_3[1].X);
00384             Y1 = (int) viewPort.PriceToPixels(PointsForDrawing_1_3[0].Y);
00385             Y2 = (int) viewPort.PriceToPixels(PointsForDrawing_1_3[1].Y);
00386             //проверка на касание трендовой
00387             if ((X > X1) && (X < X2))
00388             {
00389                 if (Math.Abs((Y - Y1) * (X2 - X1) - (X - X1) * (Y2 - Y1)) < ((Math.Abs(Y2 - Y1) + (X2 - X1)) * 2))
00390                     return true;
00391             }
00392 
00393             X1 = (int) viewPort.BarNumberToPixels(PointsForDrawing_21_4[0].X);
00394             X2 = (int) viewPort.BarNumberToPixels(PointsForDrawing_21_4[1].X);
00395             Y1 = (int) viewPort.PriceToPixels(PointsForDrawing_21_4[0].Y);
00396             Y2 = (int) viewPort.PriceToPixels(PointsForDrawing_21_4[1].Y);
00397 
00398             //проверка на касание целевой
00399             if ((X > X1) && (X < X2))
00400             {
00401                 if (Math.Abs((Y - Y1) * (X2 - X1) - (X - X1) * (Y2 - Y1)) < ((Math.Abs(Y2 - Y1) + (X2 - X1)) * 2))
00402                     return true;
00403             }
00404 
00405             return false;
