00086         public static TF getTF(int minutes)
00087         {
00088             if (minutes < 60) return TF.custom;
00089             switch (minutes)
00090             {
00091                 case 60:
00092                     return TF.m60;
00093                 case 240:
00094                     return TF.m240;
00095                 case 1440:
00096                     return TF.Day;
00097                 case 10080:
00098                     return TF.Week;
00099                 default:
00100                     if (minutes >= 28 * 1440)
00101                     {
00102                         if (minutes <= 31 * 1440) return TF.Month;
00103                         if (Math.Abs(minutes - 131490) < 3 * 1440) return TF.Quarter;
00104                         if (Math.Abs(minutes - 525960) < 2880) return TF.Year;
00105                     }
00106                     return TF.custom;
00107             }

00211         {
00212             int len = package.GetLength(0);
00213             double[] quotes = new double[len * 5];
00214             for (int i = 0, o = 0; i < len; i++, o+=5)
00215             {
00216                 long dt=0;
00217                 for (int j = 0; j < 4; j++)
00218                 {
00219                     quotes[o + j + 1] = (package[i, j] & 0x00ffffff) * pip;
00220                     dt |= (package[i, j] & 0xff000000) >> (j * 8);
00221                 }
00222                 quotes[o] = (double)dt;
00223             }
00224             return quotes;
00225         }
