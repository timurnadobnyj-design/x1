00121     public void getIndex(out int hl, out int bar37, out int frep, out int trndir)
00122     {
00123         //является ли бар high или low
00124         hl = (int)(index & 3);
00125         //является ли бар баром "3"(для 2-5точек модели) или "7" (для 1-й точкимодели
00126         bar37 = (int)((index & 60) >> 2);
00127         //репером какого фрейма является бар
00128         frep = (int)((index & 192) >> 6);
00129         //направление тренда на фрейме, на котором бар является репером
00130         trndir = (int)((index & 261888) >> 8);

00133     public string whatExtremum()
00134     {
00135         int stat37 = (int)((index & 60) >> 2);
00136         int high37 = (stat37 & 12) >> 2;//флаг для бара хай является ли он баром "3" или "7"
00137         int low37 = (stat37 & 3);//флаг для бара лоу является ли он баром "3" или "7"
00138         if ((high37 > 0) && (low37 == 0)) return "Up";
00139         if ((low37 > 0) && (high37 == 0)) return "Dn";
00140         if ((high37 > 0) && (low37 > 0)) return "UpDn";
00141         return " ";
