00364         private void GridSplitter_MouseMove(object sender, MouseEventArgs e)
00365         {
00366

00368             if (!IsLineGrabbed)

00370                 if ((e.X < leftLine.xPlusGrabZone && e.X > leftLine.xMinusGrabzone) || (e.X > rightLine.xMinusGrabzone && e.X < rightLine.xPlusGrabZone))

00394                             mouseArgs.mouseDragDelta.X = 0;
