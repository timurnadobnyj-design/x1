00394                 int low37 = (bar37 & 3);//флаг для бара лоу является ли он баром "3" или "7"
00395                 if ((high37 > 0) && (low37 == 0)) return TDirection.dtUp;
00396                 if ((low37 > 0) && (high37 == 0)) return TDirection.dtDn;
00397                 if ((high37 > 0) && (low37 > 0))
00398                 {
00399                     if (ADir == TDirection.dtUp) return TDirection.dtUp;
00400                     if (ADir == TDirection.dtDn) return TDirection.dtDn;
00401                     return TDirection.dtUpDn;

00403

00405                 return TDirection.dtUnKnown;
00406             }
00407             else 
00408             {
00409                 int F;
00410                 int L;
00411                 int I;

00417                 TDirection RHigh = TDirection.dtUnKnown;
00418                 Size = ASize;
00419                     F = Bar - Size;

00421                 while ((R == TDirection.dtUnKnown) && (F >= 0) && ((Size <= 7) || (Size <= ASize)))
00422                 {
00423                     // Просмотр баров от Bar влево до Bar-Size

00436                         R = RHigh;

00439                             R = RLow;

00441                             R = TDirection.dtUnKnown;
