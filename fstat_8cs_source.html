00126         public void CountTrianglesArea(TModel m, TQuotes q)
00127         {
00128             double pip = m.ModelDir == TDirection.dtUp ? q.Symbol.pip : -q.Symbol.pip;
00129             double l12 = Math.Sqrt(Math.Pow(m.Point2.Bar - m.Point1.Bar, 2) + Math.Pow((m.Point2.Price - m.Point1.Price) / pip, 2));
00130             double l13 = Math.Sqrt(Math.Pow(m.Point3.Bar - m.Point1.Bar, 2) + Math.Pow((m.Point3.Price - m.Point1.Price) / pip, 2));
00131             double l23 = Math.Sqrt(Math.Pow(m.Point3.Bar - m.Point2.Bar, 2) + Math.Pow((m.Point2.Price - m.Point3.Price) / pip, 2));
00132             double l24 = Math.Sqrt(Math.Pow(m.Point4.Bar - m.Point2.Bar, 2) + Math.Pow((m.Point4.Price - m.Point2.Price) / pip, 2));
00133             double l34 = Math.Sqrt(Math.Pow(m.Point4.Bar - m.Point3.Bar, 2) + Math.Pow((m.Point4.Price - m.Point3.Price) / pip, 2));
00134 
00135             double p = (l12 + l13 + l23) / 2;
00136             S123 = Math.Round(Math.Sqrt(p * (p - l12) * (p - l13) * (p - l23)));
00137             p = (l23 + l24 + l34) / 2;
00138             S234 = Math.Round(Math.Sqrt(p * (p - l23) * (p - l24) * (p - l34)));
00139             S345 = S456 = 0;
00140 
00141             if (m.Point5.Bar > m.Point4.Bar)
00142             {
00143                 double l35 = Math.Sqrt(Math.Pow(m.Point5.Bar - m.Point3.Bar, 2) + Math.Pow((m.Point5.Price - m.Point3.Price) / pip, 2));
00144                 double l45 = Math.Sqrt(Math.Pow(m.Point5.Bar - m.Point4.Bar, 2) + Math.Pow((m.Point4.Price - m.Point5.Price) / pip, 2));
00145                 p = (l34 + l35 + l45) / 2;
00146                 S345 = Math.Round(Math.Sqrt(p * (p - l34) * (p - l35) * (p - l45)));
00147 
00148                 if (m.Point6.Bar > m.Point4.Bar)
00149                 {
00150                     double l46 = Math.Sqrt(Math.Pow(m.Point6.Bar - m.Point4.Bar, 2) + Math.Pow((m.Point6.Price - m.Point4.Price) / pip, 2));
00151                     double l56 = Math.Sqrt(Math.Pow(m.Point6.Bar - m.Point5.Bar, 2) + Math.Pow((m.Point6.Price - m.Point5.Price) / pip, 2));
00152                     p = (l45 + l46 + l56) / 2;
00153                     S456 = Math.Round(Math.Sqrt(p * (p - l45) * (p - l46) * (p - l56)));
00154                 }
00155             }
00156         }
