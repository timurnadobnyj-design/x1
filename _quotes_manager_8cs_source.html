00159         public void Config_Init()
00160         {
00161             string config = XMLConfig.Get("DataSourceLibraries");
00162             if (config.Length == 0) return;
00163             //parse config string
00164             //string config = "modulename^checked>bool,prompt>string[;...]";
00165             char delim = config.IndexOf('|') > 0 ? '|' : ';';
00166             foreach (string a in config.Split(delim))
00167             {
00168                 Dictionary<string, string> child = new Dictionary<string, string>();
00169                 string[] b = a.Split('^');
00170                 if (b.Length != 2) continue;
00171                 char splitter = config.IndexOf('~') > 0 ? '~' : ',';
00172                 foreach (string c in b[1].Split(splitter))
00173                 {
00174                     string[] d = c.Split('>');
00175                     if (d.Length == 2) child.Add(d[0], d[1]);
00176                 }
00177                 Config.Add(b[0], child);
00178             }
00179         }

00185         public void Config_Save()
00186         {
00187             //join hash to string function
00188             string hash = "";
00189             foreach (string a in Config.Keys)
00190             {
00191                 hash = hash.Length == 0 ? a : hash + "|" + a;
00192                 string val = "";
00193                 foreach (string b in Config[a].Keys)
00194                 {
00195                     if (val.Length > 0) val += "~";
00196                     val += b + ">" + Config[a][b];
00197                 }
00198                 hash += "^" + val;
00199             }
00200             XMLConfig.Set("DataSourceLibraries", hash);
00201         }
